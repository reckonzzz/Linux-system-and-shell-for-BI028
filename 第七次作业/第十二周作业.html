3.1 寻找质数的函数
# 判断n是否为质数
def is_prime(n):
    if n == 1 or n <= 0:
        return False
    elif n == 2 or n == 3:
        return True
    elif n % 6 == 1 or n % 6 == 5:
        for i in range(2, n):
            if n % i == 0:
                return False
        return True
    else:
        return False


# 输出所有小于n的质数
def less(n):
    for i in range(0, n):
        if is_prime(i):
            print(i, end=' ')


# 输出前n个质数
def head(n):
    count = 0
    i = 0
    while count < n:
        if is_prime(i):
            print(i, end=' ')
            count = count + 1
        i = i + 1

def fun(n):

def root(f, a, b):
    x = (a + b) / 2.0
    if f(a) == 0:
        return a
    elif f(b) == 0:
        return b
    elif f(x) == 0:
        return x
    if abs(b-a) < 0.001:
        return x
    elif f(a) * f(b) < 0:
        if f(a) * f(x) < 0:
            return root(f, a, x)
        else:
            return root(f, x, b)
